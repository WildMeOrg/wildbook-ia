[docs]def get_normalized_score_column(fsv, colx, min_, max_, power):
    fs = fsv.T[colx].T.copy()
    fs = fs if min_ == 0 and max_ == 1 else vt.clipnorm(fs, min_, max_, out=fs)
    fs = fs if power == 1 else np.power(fs, power, out=fs)
    return fs
