>>> ungr_gids = ibs.get_encounter_gids([ungr_eid])[0]
>>> #Now let's make sure that when we recompute encounters, our non-special eid remains the same

>>> print("COMPUTE: New encounters are %r" % ibs.get_valid_eids())
